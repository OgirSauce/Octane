                        //get the values from the UI
                        //start or controller function
                        function getValues(){
                            //get values from the page
                            let startValue = document.getElementById("startValue").value;
                            let endValue = document.getElementById("endValue").value;
                        
                            //we need to validate our input
                            //parse into intergers
                            startValue = parseInt(startValue);
                            endValue = parseInt(endValue);
                        
                            if (Number.isInteger(startValue) && Number.isInteger(endValue)){
                                //we call generateNumbers
                                let numbers = generateNumbers(startValue,endValue);
                                //we call displayNumbers
                                displayNumbers(numbers);
                        
                            }else{
                                alert("You must enter integers")
                            }
                        
                        }
                        
                        //generate number from startValue to the endValue
                        //logic function(s)
                        function generateNumbers(sValue, eValue){
                        
                            let numbers = [];
                        
                            //we want to get all numbers from start to end
                        
                            for (let index = sValue; index <= eValue; index++){
                                //this will execute in a loop unil index = eValue
                                numbers.push(index);
                            }
                        
                            return numbers;
                        
                        }
                        
                        //display numbers and mark even numbers bold
                        //display or view functions
                        function displayNumbers(numbers) {
                        
                        
                            let templateRows = "";
                        
                            for (let index = 0; index < numbers.length; index++) {
                                
                                let className = "even";
                                let number = numbers[index];
                        
                                if(number % 2 == 0){
                                    className = "even";
                                }
                                else{
                                    className = "odd";
                                }
                                //This does render correctly with Prisim see the source
                                templateRows += `${number}`
                                
                            }
                        
                            document.getElementById("results").innerHTML = templateRows
                        
                        }

This code is structured in three functions.

Octane

Octane is a function that allows you to select a string of numbers from 0-100. It will display the numbers in a row with the even numbers in bold.

function getValues; is coded to obtain the starting and ending values inputs. The function converts those inputs into intergers using the perseINT funtion. If the user inputs anything other then an integer. There is an alert that will inform the user to place an integer.

function generateNumbers; is coded to generate the integers that need to apear acording to the inputs that where entered.

function displayNumbers; is coded to display the generated numbers. It's coded with class names to seperate the even and odd numbers. I use these class names in the css to assign a color to the end results. In this project I choose to display the even numbers in bold.
